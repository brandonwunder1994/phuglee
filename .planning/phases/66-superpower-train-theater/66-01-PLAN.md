```js
// public/js/bridge-train.js
function countOpenTrainGroups(data, decidedKeys) {
  var groups = getReviewGroups(data);
  var all = (groups.distressed || []).concat(groups.notDistressed || []);
  return filterUndecidedTrainGroups(all, decidedKeys).length;
}
// Export on BridgeTrain.countOpenTrainGroups
```

<!-- Pivot target (bridge.js) -->
```js
let forceTrainTheater = false;

// processUpload success (existing clearTrainDecidedKeys block):
clearTrainDecidedKeys();
// ... existing search/page reset ...
forceTrainTheater = true;
renderResults(data);

// renderResults admin branch — REPLACE:
//   setResultsMode(resultsMode || 'kept');
// WITH:
const openCount = (window.BridgeTrain && typeof BridgeTrain.countOpenTrainGroups === 'function')
  ? BridgeTrain.countOpenTrainGroups(data, trainDecidedKeys)
  : filterUndecidedTrainGroups(
      (getReviewGroups(data).distressed || []).concat(getReviewGroups(data).notDistressed || [])
    ).length;
updateTrainMissionHeader(openCount, (data.rows || []).length);
if (forceTrainTheater) {
  forceTrainTheater = false;
  setResultsMode(openCount > 0 ? 'train' : 'kept');
} else {
  setResultsMode(resultsMode || 'kept');
}
// Optional: trainWrap.classList.toggle('is-theater', openCount > 0 && isBridgeAdmin());
```
