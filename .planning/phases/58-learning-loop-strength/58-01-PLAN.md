```js
/** Gold address fragments (match Phase 57 fixtures) */
const GOLD_KEEP_FRAGS = [
  '100 Gold Keep', '200 Gold Keep', '300 Gold Keep',
  '400 Gold Keep', '500 Gold Keep'
];
const GOLD_DENY_FRAGS = [
  '100 Gold Deny', '200 Gold Deny', '300 Gold Deny',
  '400 Gold Deny', '500 Gold Deny'
];

/** Same ban as ACC-02 gold suite */
const BANNED_SILENT_DROP_RE = /no_type(_column)?|unresolved_map|missing_type|low_confidence|cleaner|silent/i;

/**
 * @param {Array<{action?:string, at?:string}>} events
 * @param {{ windowSize?: number }} [opts]  default windowSize=25 (last N vs previous N)
 * @returns {{
 *   recentCount: number,
 *   previousCount: number,
 *   direction: 'down'|'up'|'flat',
 *   window: string,  // e.g. 'last_25_vs_prior_25'
 *   decisionActionsOnly: true
 * }}
 * Only counts actions in DECISION_ACTIONS. Ignores undo, phrase audits, process_apply.
 * direction: recent < previous → 'down'; recent > previous → 'up'; else 'flat'.
 */
function computeDecisionTrend(events, opts)

/**
 * Binary keep/kill scorer against processUpload-shaped result.
 * Strong = distressedSignalTag === 'Strong Distressed Signal' OR includes 'Strong Distressed'
 * TP: keep frag Strong in result.rows
 * FN: keep frag not Strong kept
 * FP: deny frag Strong in result.rows
 * TN: deny frag not Strong kept
 * precision = TP/(TP+FP) or 1 if denom 0
 * recall = TP/(TP+FN) or 1 if denom 0
 *
 * @param {object} result  processUpload result { rows, notDistressedRows?, discarded?, stats? }
 * @param {{ keepFrags?: string[], denyFrags?: string[] }} [opts]
 * @returns {{ tp, fp, fn, tn, precision, recall, keepFrags, denyFrags }}
 */
function scoreGoldKeepKill(result, opts)

/**
 * Apply coverage from process meta / applied ids.
 * @param {{
 *   appliedRuleIds?: string[],
 *   activeRuleIdsExpected?: string[],  // if provided, sufficient requires intersection non-empty
 *   rowHitRate?: number|null,
 *   source?: string
 * }} input
 * @returns {{
 *   appliedRuleIds: string[],
 *   rulesAppliedCount: number,
 *   rowHitRate: number|null,
 *   sufficient: boolean,
 *   note: string,
 *   source: string
 * }}
 * Rules:
 * - sufficient=true only if rulesAppliedCount > 0 when activeRuleIdsExpected is empty/omitted
 *   OR intersection(applied, expected) non-empty when expected provided
 * - empty applied + expected=[] → sufficient=false with note that coverage needs process apply evidence
 * - never accepts groupsHidden / hiddenGroupCount as inputs (do not define those fields)
 */
function scoreApplyCoverage(input)

/**
 * Composite learning health.
 * @param {{
 *   events?: array,
 *   goldScore?: { precision, recall, baselinePrecision?, baselineRecall? }|null,
 *   applyCoverage?: object|null,  // raw input for scoreApplyCoverage OR already scored
 *   baseline?: { precision: number, recall: number },  // default {1,1} for synthetic gold
 *   epsilon?: number  // default 0
 *   discardBlob?: string  // optional joined discard reasons — if BANNED_SILENT_DROP_RE matches, force fail
 * }} input
 * @returns {{
 *   decisionTrend: ReturnType<computeDecisionTrend>,
 *   gold: {
 *     precision: number|null,
 *     recall: number|null,
 *     baselinePrecision: number,
 *     baselineRecall: number,
 *     degraded: boolean,
 *     source: string
 *   },
 *   applyCoverage: ReturnType<scoreApplyCoverage>,
 *   silentDropFailure: boolean,
 *   pairedOk: boolean
 * }}
 * pairedOk = !silentDropFailure && !gold.degraded && applyCoverage.sufficient
 *   (when goldScore null: gold.degraded=false, precision/recall null, source='unavailable' —
 *    pairedOk still requires applyCoverage.sufficient for LRN-02 gate when scoring coverage)
 * When goldScore provided: degraded if precision < baselineP - eps OR recall < baselineR - eps
 */
function buildLearningHealth(input)
```
